2. **`test_suites` table:**
   - `id` text PK
   - `userId` text NOT NULL, FK to users.id (onDelete cascade)
   - `url` text NOT NULL
   - `featureDescription` text NOT NULL
   - `status` text NOT NULL default 'pending' (values: pending, generating, complete, failed)
   - `testCaseCount` integer NOT NULL default 0
   - `createdAt` timestamp NOT NULL defaultNow()
   - `updatedAt` timestamp NOT NULL defaultNow()

3. **`test_cases` table:**
   - `id` text PK
   - `suiteId` text NOT NULL, FK to test_suites.id (onDelete cascade)
   - `name` text NOT NULL
   - `description` text NOT NULL
   - `category` testCaseCategoryEnum NOT NULL
   - `priority` testCasePriorityEnum NOT NULL
   - `reasoning` text (nullable)
   - `steps` jsonb (nullable -- populated after step generation)
   - `testRunId` text (nullable -- populated when test case is executed, FK to test_runs.id with onDelete set null)
   - `order` integer NOT NULL
   - `createdAt` timestamp NOT NULL defaultNow()

4. **`accessibility_results` table:**
   - `id` text PK
   - `resultId` text NOT NULL, FK to test_run_results.id (onDelete cascade)
   - `violationCount` integer NOT NULL default 0
   - `passCount` integer NOT NULL default 0
   - `incompleteCount` integer NOT NULL default 0
   - `inapplicableCount` integer NOT NULL default 0
   - `violations` jsonb NOT NULL default '[]' (Array of {id, impact, description, help, helpUrl, tags, nodes: [{target, html, impact, failureSummary}], nodeCount})
   - `createdAt` timestamp NOT NULL defaultNow()

4. After `executeSteps()` returns (after the step event publishing loop, before stopping screencast), add axe-core scanning:
```typescript
// Run accessibility scan on final page state (best-effort)
let accessibilityData: AccessibilityData | null = null;
try {
  const axeResults = await new AxeBuilder({ page })
    .withTags(['wcag2a', 'wcag2aa', 'wcag21a', 'wcag21aa'])
    .analyze();

  accessibilityData = {
    violationCount: axeResults.violations.length,
    passCount: axeResults.passes.length,
    incompleteCount: axeResults.incomplete.length,
    inapplicableCount: axeResults.inapplicable.length,
    violations: axeResults.violations.map(v => ({
      id: v.id,
      impact: v.impact ?? null,
      description: v.description,
      help: v.help,
      helpUrl: v.helpUrl,
      tags: v.tags,
      nodes: v.nodes.slice(0, 10).map(n => ({
        target: n.target as string[],
        html: n.html.substring(0, 500),
        impact: n.impact ?? null,
        failureSummary: n.failureSummary,
      })),
      nodeCount: v.nodes.length,
    })),
  };
} catch {
  // Accessibility scanning is best-effort -- never break test execution
}
```
