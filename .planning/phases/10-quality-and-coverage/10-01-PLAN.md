           'packages/web/src/**/*.tsx',

           '**/node_modules/**',
           '**/__tests__/**',
           '**/__test-utils__/**',
           '**/*.test.{ts,tsx}',
           '**/dist/**',
           '**/index.ts',
           '**/types/**',
           '**/*.d.ts',
           'packages/web/src/routeTree.gen.ts',

           'packages/web/src/styles.css',
           'packages/web/src/logo.svg',

           functions: 50,

Cast to `import('playwright').Locator`. Both functions accept `overrides?: Partial<Record<string, unknown>>` spread into the mock object before casting.
